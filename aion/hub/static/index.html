Data Processing

from aion.data import load_csv, train_val_test_split, augment_text rows = load_csv("data.csv") train, val, test = train_val_test_split(rows, seed=42) variants = augment_text("hello world", num_variants=4)

Tokenization

from aion.tokenizer import BPETokenizer bpe = BPETokenizer(vocab_size=8000) bpe.train(["your corpus here..."] * 100) ids = bpe.encode("hello world") text = bpe.decode(ids)

Caching

from aion.cache import MemoryCache, cached cache = MemoryCache(default_ttl=300) cache.set("key", {"value": 42}) cache.get("key") @cached(ttl=60) def expensive(x): return x ** 2

Data Structures

from aion.structures import Trie, BloomFilter, LRUCache, UnionFind trie = Trie() trie.insert("python") trie.starts_with("py") # ["python"] bf = BloomFilter(100000) bf.add("item") bf.might_contain("item") # True

Pipelines

from aion.pipeline import Pipeline, MapStep, FilterStep pipe = Pipeline([ MapStep("double", lambda x: x * 2), FilterStep("positive", lambda x: x > 0), ]) result = pipe.execute([1, -2, 3, -4, 5]) # [2, 6, 10]

Agents (requires API key)

from aion.agents import ReActAgent from aion.providers import OpenAIProvider from aion.tools import ToolRegistry, function_tool registry = ToolRegistry() registry.register("double", lambda n: n * 2) tools = [function_tool("double", "Double a number", ...)] agent = ReActAgent(provider=OpenAIProvider(), registry=registry, tools=tools) answer = agent.run("What is 21 doubled?")

Experiment Tracking

from aion.tracker import Tracker tracker = Tracker(".experiments") run = tracker.start_run("baseline") run.log_params({"lr": 0.001}) run.log_metric("loss", 0.42) run.end() tracker.compare_runs(metric_name="loss")
